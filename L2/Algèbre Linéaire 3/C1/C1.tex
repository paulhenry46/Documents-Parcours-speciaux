\begin{definition}[Produit scalaire]
    Soit $E$ un ev sur $\R$ de dimebsion finie ou non.
    $$\varphi : E\times E \to \R, (x,y)\to \varphi(x,y)$$

        \item $\varphi$ est symÃĐtriuque $\iff \varphi(x,y) = \varphi(y,x)$
        \item $\forall x, \varphi(x,x) \geq0$
        \item $\varphi$ est dÃĐfinie positive $\iff \forall x, \varphi(x,x)\geq 0 \wedge \varphi(x,x) = 0 \Rightarrow x=0$
    \end{itemize}
    On dit que $\varphi$ est un produit scalaire $\iff \varphi$ est une forme bilinÃĐaire symÃĐtrique dÃĐfinie positive.
\end{definition}
