\chapter{Second principe de thermodynamique}


\section{Nécessité d'un second principe}

\checkInfo{Contexte}{On considère deux compartiments rigides et indéformables ($A$ et $B$), initialement à des températures différentes, séparés par une paroi fixe mais perméable à la chaleur (diathermane).}

D'après le premier principe appliqué à chaque compartiment (parois fixes $\Rightarrow W=0$) :
$$\Delta U^A = Q^A \quad \text{et} \quad \Delta U^B = Q^B$$
Pour le système isolé $\{A+B\}$, l'énergie interne se conserve :
$$\Delta U^{A+B} = \Delta U^A + \Delta U^B = 0 \implies Q^A = -Q^B$$
Le premier principe n'impose aucune contrainte sur le signe de $Q^A$. C'est le second principe qui formalise le sens de l'évolution spontanée.

\begin{theorem}[Énoncé de Clausius]
    La chaleur ne peut pas passer spontanément d'un corps froid à un corps chaud sans apport de travail extérieur.
\end{theorem}

\section{Cycle de Carnot (Ditherme)}

Un cycle de Carnot est un cycle idéal composé de 4 transformations réversibles (quasi-statiques et sans frottements) :
\begin{enumerate}
    \item Deux isothermes aux températures $T_{ch}$ (chaude) et $T_{fr}$ (froide).
    \item Deux adiabatiques reliant ces isothermes.
\end{enumerate}

\begin{definition}[Efficacité / Rendement]
    L'efficacité $\epsilon$ d'une machine thermique est le rapport de l'énergie utile sur l'énergie coûteuse :
    $$\epsilon = \left| \frac{\text{Énergie utile}}{\text{Énergie coûteuse}} \right|$$
\end{definition}

\warningInfo{Cas du moteur}{Pour un moteur, l'énergie utile est le travail fourni $|W_{cycle}|$ et l'énergie coûteuse est la chaleur prélevée à la source chaude $Q_{ch}$.
$$\epsilon_{mot} = \frac{-W_{cycle}}{Q_{ch}} = \frac{Q_{ch} + Q_{fr}}{Q_{ch}} = 1 + \frac{Q_{fr}}{Q_{ch}}$$}


\subsection{Rendement de Carnot}
Pour un cycle de Carnot (entièrement réversible), en utilisant les lois de Laplace sur les adiabatiques et l'expression du transfert thermique sur les isothermes ($Q = nRT \ln(V_f/V_i)$), on démontre que :
$$\frac{Q_{fr}}{Q_{ch}} = -\frac{T_{fr}}{T_{ch}}$$

\begin{proposition}[Théorème de Carnot]
    L'efficacité d'un moteur ditherme est maximale si le cycle est réversible. Cette efficacité maximale, dite de Carnot, ne dépend que des températures des sources :
    $$\epsilon_{max} = 1 - \frac{T_{fr}}{T_{ch}}$$
\end{proposition}

\criticalInfo{Inégalité de Clausius}{Pour tout cycle réel (irréversible), $\epsilon \leq \epsilon_{max}$, ce qui conduit à la relation fondamentale :
$$\frac{Q_{ch}}{T_{ch}} + \frac{Q_{fr}}{T_{fr}} \leq 0$$}

\section{Généralisation : L'Entropie}

Pour un cycle quelconque quasi-statique (QS), l'inégalité devient l'intégrale de Clausius :
$$\oint \frac{\delta Q_{rev}}{T} = 0$$

\begin{theorem}
    Comme l'intégrale de $\frac{\delta Q_{rev}}{T}$ sur un cycle fermé est nulle, la quantité $\int_A^B \frac{\delta Q_{rev}}{T}$ est indépendante du chemin suivi. Elle définit la variation d'une fonction d'état appelée \textbf{Entropie}, notée $S$.
    $$\Delta S_{AB} = S_B - S_A = \int_A^B \frac{\delta Q_{rev}}{T}$$
\end{theorem}

\tipsInfo{Calcul de $\Delta S$}{L'entropie étant une fonction d'état, pour calculer $\Delta S$ d'une transformation irréversible, on imagine un chemin réversible entre le même état initial et final.}

\section{Énoncé du second principe}
\begin{theorem}[Énoncé détaillé]
    $$\Delta_{i\to f} = S_f-S_i = \int_i^f \frac{\delta Q_{QS}}{T} = -\Delta S_{ext, if}+S^p_{i\to f}$$
    avec 
    \begin{itemize}
        \item $-\Delta S_{ext,if} = 0$ si la transformation est adiabatique
        \item $-\Delta S_{ext,if} = \frac{Q_{i\to f}}{T_{source}}$ si la transformation s'effectue au contact d'une source.
    \end{itemize}
      et
      \begin{itemize}
        \item $S^p_{if}=0$ si c'est réversible
        \item $S^p_{if}>0$ si c'est irréversible
      \end{itemize}
\end{theorem}
\tipsInfo{Signification physique}{  $S_e = \int \frac{\delta Q}{T_{ext}}$ est l'entropie échangée avec le milieu extérieur.
        et $S_p$ est l'entropie produite (ou créée) au sein du système.
    }

\begin{theorem}[Entropie de l'Univers]
    Pour un système isolé (ou l'Univers entier), $Q=0 \implies S_e = 0$.
    Par conséquent : $\Delta S_{univers} \geq 0$.
    L'entropie d'un système isolé ne peut que croître.
\end{theorem}

\section{Énoncé de Thomson / de Kelvin}

\criticalInfo{Rappel : Moteur}{Pour un cycle moteur, le travail total échangé est négatif ($W < 0$), ce qui signifie que le système fournit de l'énergie mécanique au milieu extérieur.}

\begin{theorem}[Énoncé de Thomson]
    Un système en contact avec une seule source de chaleur (cycle monotherme) ne peut pas fournir de travail au milieu extérieur au cours d'un cycle. 
    Autrement dit : pour un cycle monotherme, $W \geq 0$.
\end{theorem}

\begin{myproof}[Démonstration par le second principe]
    Soit un cycle monotherme en contact avec une source à la température $T_0$.
    \begin{itemize}
        \item Premier principe : $\Delta U = W + Q = 0 \implies W = -Q$.
        \item Second principe : $\Delta S = \frac{Q}{T_0} + S^p = 0 \implies Q = -T_0 S^p$.
    \end{itemize}
    En combinant les deux, on obtient : $W = T_0 S^p$.
    Comme $S^p \geq 0$ et $T_0 > 0$, alors $W \geq 0$. Le travail ne peut pas être fourni par le système.
\end{myproof}

\begin{theorem}[Conséquence]
    Il est impossible de transformer intégralement de la chaleur en travail de façon cyclique. En revanche, la transformation intégrale de travail en chaleur est possible (ex: effet Joule).
\end{theorem}

\section{Exemples d'application}

\subsection{Expérience 1 : Compression monotherme brusque}

\checkInfo{Dispositif}{Un cylindre contenant un gaz parfait (GP) est placé dans un thermostat à $T_0$. On pose brusquement une masse sur le piston, faisant passer la pression de $P_i$ à $P_f = \alpha P_i$ (avec $\alpha > 1$).}



1. Entropie échangée $S^e$ :
La transformation est monotherme ($T_{ext} = T_0$) et le gaz est parfait ($\Delta U = 0$ car $T_i = T_f$).
$$Q = -W = \int P_{ext} dV = P_f (V_f - V_i)$$
En utilisant $V = nRT_0/P$, on a : $V_f - V_i = \frac{nRT_0}{P_f} - \frac{nRT_0}{P_i} = \frac{nRT_0}{P_f} (1 - \alpha)$.
D'où $Q = nRT_0 (1 - \alpha)$.
$$S^e = \frac{Q}{T_0} = nR(1 - \alpha)$$

2. Variation d'entropie $\Delta S$ :
On utilise un chemin réversible (isotherme QS) pour le calcul :
$$\Delta S = \int_i^f \frac{\delta Q_{rev}}{T} = \int_i^f \frac{P dV}{T} = nR \ln\left(\frac{V_f}{V_i}\right) = nR \ln\left(\frac{1}{\alpha}\right)$$

3. Entropie produite $S^p$ :
$$S^p = \Delta S - S^e = nR \left[ \ln\left(\frac{1}{\alpha}\right) + \alpha - 1 \right]$$

\tipsInfo{Analyse}{La fonction $f(\alpha) = \ln(1/\alpha) + \alpha - 1$ est toujours positive pour $\alpha \neq 1$ et s'annule pour $\alpha = 1$ (cas réversible). $S^p$ mesure ici l'irréversibilité liée au déséquilibre de pression.}

\subsection{Calcul de $\Delta S$ entre deux états quelconques}

L'entropie étant une fonction d'état, on peut choisir n'importe quel chemin fictif quasi-statique entre $(P_0, V_0, T_0)$ et $(P_1, V_1, T_1)$.

\begin{proposition}[Formules de l'entropie du Gaz Parfait]
    Selon les variables d'état choisies, on utilise les identités thermodynamiques :
    \begin{itemize}
        \item \textbf{Variables $(T, V)$ :} $dS = \frac{dU + P dV}{T} = C_v \frac{dT}{T} + nR \frac{dV}{V}$
        $$\Delta S = C_v \ln\left(\frac{T_1}{T_0}\right) + nR \ln\left(\frac{V_1}{V_0}\right)$$
        \item \textbf{Variables $(T, P)$ :} $dS = \frac{dH - V dP}{T} = C_p \frac{dT}{T} - nR \frac{dP}{P}$
        $$\Delta S = C_p \ln\left(\frac{T_1}{T_0}\right) - nR \ln\left(\frac{P_1}{P_0}\right)$$
    \end{itemize}
\end{proposition}

\subsection{Démonstration du chemin par étapes}

On souhaite aller de l'état $0$ $(P_0, V_0, T_0)$ à l'état $1$ $(P_1, V_1, T_1)$. On définit un état intermédiaire $A$ tel que :
\begin{itemize}
    \item Étape 1 ($0 \to A$) : Transformation isochore ($V_A = V_0$). Pour atteindre la pression $P_1$, la température devient $T_A = \frac{P_1 V_0}{nR}$.
    \item Étape 2 ($A \to 1$) : Transformation isobare ($P_A = P_1$). On fait varier le volume de $V_0$ à $V_1$.
\end{itemize}



\subsubsection{Calcul des variations d'entropie}

1. Sur l'isochore ($0 \to A$) :
   Comme $dV = 0$, l'identité thermodynamique $dU = TdS - PdV$ devient $dU = TdS$.
   $$\Delta S_{0A} = \int_{T_0}^{T_A} \frac{C_v dT}{T} = C_v \ln\left(\frac{T_A}{T_0}\right)$$

2. Sur l'isobare ($A \to 1$) :
   Comme $dP = 0$, l'identité $dH = TdS + VdP$ devient $dH = TdS$.
   $$\Delta S_{A1} = \int_{T_A}^{T_1} \frac{C_p dT}{T} = C_p \ln\left(\frac{T_1}{T_A}\right)$$

3. Somme totale :
   $$\Delta S_{total} = C_v \ln\left(\frac{T_A}{T_0}\right) + C_p \ln\left(\frac{T_1}{T_A}\right)$$


\subsubsection{Récupération de la formule classique}

On utilise la relation de Mayer $C_p = C_v + nR$ pour décomposer le second terme :
\begin{flalign*}
    \Delta S &= C_v \ln\left(\frac{T_A}{T_0}\right) + (C_v + nR) \ln\left(\frac{T_1}{T_A}\right)\\
&= C_v \left[ \ln\left(\frac{T_A}{T_0}\right) + \ln\left(\frac{T_1}{T_A}\right) \right] + nR \ln\left(\frac{T_1}{T_A}\right)\\
&= C_v \ln\left(\frac{T_1}{T_0}\right) + nR \ln\left(\frac{T_1}{T_A}\right)
\end{flalign*}

Or, sur l'isobare $A \to 1$, on a $\frac{V}{T} = const \implies \frac{T_1}{T_A} = \frac{V_1}{V_A}$. Comme $V_A = V_0$ par construction :
$$\Delta S = C_v \ln\left(\frac{T_1}{T_0}\right) + nR \ln\left(\frac{V_1}{V_0}\right)$$

\tipsInfo{Conclusion}{On retrouve exactement la formule fondamentale en variables $(T, V)$. Cela prouve que peu importe la complexité du chemin quasi-statique choisi (isochore, isobare, isotherme), le résultat pour $\Delta S$ sera identique car il ne dépend que des points de départ et d'arrivée.}
